const mongoose = require('mongoose');
const Schema = mongoose.Schema;

const mascotaSchema = new Schema({
  nombre:  String
});
const saludoSchema = new Schema({
    saludo:  String,
    nose:{
        type:String,
        required: true
    }
});

const Mascota = mongoose.model('mascota', mascotaSchema);
